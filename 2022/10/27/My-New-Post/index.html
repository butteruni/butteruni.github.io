这是一个并查集的模板

#include <stdio.h>
int bin[1002];
int findx(int x) //返回一个元素的父元素;
{
int r = x;
while(bin[r] != r){r = bin[r];}
return r;
}
void merge(int x,int y)
{
int fx,fy;
fx = findx(x); //找这两个元素的父元素;
fy = findx(y);
if (fx != fy ){bin[fx] = fy;} //若两个元素的父元素不一样，则令一个元素成为另一元素父元素(一样则成环);
}
int main()
{
int n,m,i,x,y,count;
while(scanf(“%d”,&n),n)
{
for ( i = 1; i <= n; i++){bin[i] = i;} //先令所有元素成为自身父元素(先假定所有元素均为根元素);
for ( scanf(“%d”,&m);m > 0;m–)
{
scanf(“%d %d”,&x,&y); //读入两个相连元素;
merge(x,y);
} //此循环结束后形成x个树(即x个存在一个根元素的集合)；
for (count = -1, i = 1;i <= n; i++)
{if(bin[i] == i) count++;} //若有n个根元素，则返回n - 1;
printf(“%d\n”,count);
}
return 0;
}
